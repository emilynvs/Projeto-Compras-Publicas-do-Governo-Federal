create database salas_ancine;

use salas_ancine;

create table exibidor(
    id_exibidor int AUTO_INCREMENT  PRIMARY KEY,
    nome_exibidor VARCHAR(255),
    registro_exibidor VARCHAR(255),
    cnpj_exibidor VARCHAR(255),
    situacao_exibidor VARCHAR(255),
    nome_grupo_exibidor VARCHAR(255),
    operacao_usual VARCHAR(255)
);

create table endereco(
    id_endereco int AUTO_INCREMENT PRIMARY KEY,
    numero_endereco_complexo int,
    complemento_complexo VARCHAR(255),
    bairro_complexo VARCHAR(255),
    municipio_complexo VARCHAR(255),
    cep_complexo VARCHAR(255),
    uf_complexo VARCHAR(255),
    endereco_complexo VARCHAR(255),
    numero_complexo int,
    logradouro_complexo VARCHAR(255),
    resto_complexo VARCHAR(255)
);

create table complexo(
    id_complexo int AUTO_INCREMENT PRIMARY KEY,
    id_exibidor int,
    FOREIGN KEY (id_exibidor) REFERENCES exibidor(id_exibidor),
    id_endereco int,
    FOREIGN KEY (id_endereco) REFERENCES endereco(id_endereco),
    nome_complexo VARCHAR(255),
    complexo_itinerante VARCHAR(255),
    situacao_complexo VARCHAR(255),
    data_situacao_complexo VARCHAR(255),
    registro_complexo VARCHAR(255),
    pagina_eletronica_complexo VARCHAR(255)
);

create table sala(
    id_sala int AUTO_INCREMENT PRIMARY KEY,
    id_complexo int,
    FOREIGN KEY (id_complexo) REFERENCES complexo(id_complexo),
    registro_sala int,
    cnpj_sala VARCHAR(255),
    situacao_sala VARCHAR(255),
    data_situacao_sala VARCHAR(255),
    data_inicio_funcionamento_sala VARCHAR(255),
    assentos_sala int,
    assentos_cadeirantes int,
    assentos_mobilidade_reduzida int,
    assentos_obesidade int,
    acesso_assentos_com_rampa VARCHAR(255),
    acesso_sala_com_rampa VARCHAR(255),
    banheiros_acessiveis VARCHAR(255)
);
